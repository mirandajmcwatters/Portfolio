        setosa = iris[iris$Species == "setosa", c("Sepal.Length", "Sepal.Width", "Petal.Length", "Petal.Width")]
        versicolor = iris[iris$Species == "versicolor", c("Sepal.Length", "Sepal.Width", "Petal.Length", "Petal.Width")]
        virginica = iris[iris$Species == "virginica", c("Sepal.Length", "Sepal.Width", "Petal.Length", "Petal.Width")]
        setosa.means=c(mean(setosa$Sepal.Length),mean(setosa$Sepal.Width), mean(setosa$Petal.Length), mean(setosa$Petal.Width))
        versicolor.means=c(mean(versicolor$Sepal.Length),mean(versicolor$Sepal.Width), mean(versicolor$Petal.Length), mean(versicolor$Petal.Width))
        virginica.means=c(mean(virginica$Sepal.Length),mean(virginica$Sepal.Width), mean(virginica$Petal.Length), mean(virginica$Petal.Width))
        all.means=c(setosa.means,versicolor.means,virginica.means)
        means.matrix=matrix(all.means,nrow=3, byrow = TRUE)
        colnames(means.matrix)=c("Sepal Length", "Sepal Width", "Petal Length", "Petal Width")
        rownames(means.matrix)=c("Setosa","Versicolor","Virginica")
        print(means.matrix)

Line graphs

            
        library(forecast)
        library(ggplot2)
        ship.df = read.csv("ApplianceShipments.csv")
        summary(ship.df)
        ship.df
        ship.ts
        ship.ts=ts(ship.df$Shipments, start = c(1985,1),end = c(1989,4), freq=4)
        ship.lm = tslm(ship.ts ~ trend + I(trend^2))
        plot(ship.ts, xlab= "year", ylab="Shipments", ylim = c(3500,5000))

        #more line graphs
        par(oma = c(0,0,0,2))
        xrange <-c(1,5)
        yrange=range(ship.ts)
        plot(xrange,  yrange,  main  =  "Shipments  by  Quarter",  type  =  "n",  xlab  =  "Year",  ylab  = "Shipments", bty = "l")
        colors <-rainbow(4)
        linetype <-c(1:4)
        plotchar <-c(1:4)
        for (i in 1:4) {
          current_quarter <- subset(ship.ts, cycle(ship.ts)==i)
          lines(current_quarter, type="b", lwd=1.5, lty=linetype[i], col=colors[i], pch=plotchar[i])
        }  
        legend(5.25 ,4800,   1:4,   cex=0.8,   col=colors,   pch=plotchar,   lty=linetype,   title="Quarter", xpd=NA)
